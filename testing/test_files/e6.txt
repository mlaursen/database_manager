CREATE OR REPLACE VIEW TEST_PERSON_VIEW AS SELECT P.ID,
		LAST_NAME||', '||FIRST_NAME PERSON_NAME,
		FIRST_NAME,
		LAST_NAME,
		SALARY PERSON_SALARY,
		TYPE,
		J.NAME JOB_NAME,
		DESCRIPTION JOB_DESCRIPTION
	FROM TEST_PERSON P
	INNER JOIN TEST_JOB J ON P.JOB_ID = J.ID
	ORDER BY LAST_NAME, FIRST_NAME
WITH READ ONLY;
