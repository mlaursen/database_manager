CREATE OR REPLACE VIEW PERSON_VIEW AS SELECT P.ID,
		LAST_NAME||', '||FIRST_NAME PERSON_NAME,
		FIRST_NAME,
		LAST_NAME,
		SALARY PERSON_SALARY,
		TYPE,
		J.NAME JOB_NAME,
		DESCRIPTION JOB_DESCRIPTION
	FROM PERSON P
	INNER JOIN JOB J ON P.JOB_ID = J.ID
	ORDER BY LAST_NAME, FIRST_NAME
WITH READ ONLY;
