CREATE OR REPLACE PACKAGE BODY TEST_TEMP_ACCOUNT_PKG AS
  PROCEDURE NEW( PUSER IN TEST_TEMP_ACCOUNT.USERNAME%TYPE
               , PPASS IN TEST_TEMP_ACCOUNT.PASSWORD%TYPE
               , PCODE IN TEST_TEMP_ACCOUNT.CODE%TYPE
               , PID   IN TEST_TEMP_ACCOUNT.ID%TYPE DEFAULT SEQ_TEST_TEMP_ACCOUNT_ID.NEXTVAL
               )
  IS
    IUSER TEST_TEMP_ACCOUNT.USERNAME%TYPE;
  BEGIN
    INSERT INTO TEST_TEMP_ACCOUNT(ID, USERNAME, PASSWORD, CODE, CREATED)
    SELECT PID, PUSER, PPASS, PCODE, SYSDATE FROM DUAL
    WHERE NOT EXISTS(SELECT NULL FROM TEST_TEMP_ACCOUNT TA, TEST_ACCOUNT A WHERE TA.USERNAME=PUSER OR A.USERNAME=PUSER);
    COMMIT;
    
    EXCEPTION
      WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
  END NEW;
  
  PROCEDURE NEWACCOUNT(PID IN TEST_TEMP_ACCOUNT.ID%TYPE)
  IS
  BEGIN
    TEST_ACCOUNT_PKG.CREATEFROMTEMP(PID);
  END NEWACCOUNT;
  
  PROCEDURE NEWACCOUNT( PUSER IN TEST_TEMP_ACCOUNT.USERNAME%TYPE
                      , PPASS IN TEST_TEMP_ACCOUNT.PASSWORD%TYPE
                      )
  IS
    IID INTEGER;
  BEGIN
    SELECT ID INTO IID FROM TEST_TEMP_ACCOUNT WHERE USERNAME=PUSER AND PASSWORD=PPASS;
    NEWACCOUNT(IID);
  END NEWACCOUNT;
  
  PROCEDURE DEL(PID IN TEST_TEMP_ACCOUNT.ID%TYPE)
  IS
  BEGIN
    DELETE FROM TEST_TEMP_ACCOUNT
    WHERE ID=PID;
    COMMIT;
    
    EXCEPTION
      WHEN OTHERS THEN
        ROLLBACK;
  END DEL;
  
  PROCEDURE GET(PCURSOR OUT SYS_REFCURSOR)
  IS
  BEGIN
    OPEN PCURSOR FOR
      SELECT *
      FROM TEST_TEMP_ACCOUNT;
  END GET;
  
  PROCEDURE GET(PID IN TEST_TEMP_ACCOUNT.ID%TYPE, PCURSOR OUT SYS_REFCURSOR)
  IS
  BEGIN
    OPEN PCURSOR FOR
      SELECT ID, USERNAME, PASSWORD, CODE
      FROM TEST_TEMP_ACCOUNT
      WHERE ID=PID;
  END GET;
  
END TEST_TEMP_ACCOUNT_PKG;
